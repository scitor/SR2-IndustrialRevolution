Subsystem: IndustrialHull
	Name: #S_CIVILIAN_HULL
	Description: #S_CIVILIAN_HULL_DESC
	Tags: DefaultUnlock, NoCore, HullSystem
	Hull: Industrial
	EvaluationOrder: -100

	HexLimit := 60
	FreeHexLimit := 0

	GridWidth := 19
	GridHeight := 16

	Ship.ExternalHexes := 0
	Ship.HexSize := ShipSize / (HexLimit - FreeHexLimit)

	Repair := (ShipSize / HexLimit) * 4

	Hex.LaborCost := 0.07 * ShipSize
	Hex.BuildCost := 0.75 * ShipSize

	AddShipModifier: DamageResistFactor(0.25)
	AddShipModifier: SupplyCostFactor(0)
	//AddShipModifier: RepairSupplyCostFactor(0)
	AddShipModifier: MaintCostFactor(0)
	AddShipModifier: ThrustFactor(2.0)
	AddShipModifier: TurnThrustFactor(2.0)
	AddShipModifier: CargoStorageFactor(5.0)

	SupportCapacity := 3

	Hex.Mass := 0
	AddPostModifier: AddMass(ShipSize * (1.0 - max((ShipTotalHexes - Ship.ExternalHexes) / HexLimit, 0.0)))

	OnCheckErrors: designs::checkGlobalDesign

	Assert: ShipTotalHexes - Ship.ExternalHexes < Sum.HexLimit + 0.001
		Message: #ERROR_HEX_COUNT

	Assert: ShipSize <= 16
		Message: #ERROR_MAX_SIZE:$1:16

	Assert: TagCount.Weapon == 0
		Message: #ERROR_CANNOT_HAVE_WEAPONS

	Assert: Ship.Command > 0
		Message: #ERROR_NEED_CONTROL

	Modifier: IncreaseHexLimit(Hexes)
		HexLimit := HexLimit + Hexes
		FreeHexLimit := FreeHexLimit + Hexes
